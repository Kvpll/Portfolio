Player Health System

local HealthModule = {}
HealthModule.__index = HealthModule

function HealthModule.new(player, maxHealth)
    local self = setmetatable({}, HealthModule)
    self.player = player
    self.maxHealth = maxHealth
    self.currentHealth = maxHealth
    self.healthChanged = Instance.new("BindableEvent")
    return self
end

function HealthModule:takeDamage(amount, damageType)
    self.currentHealth = math.max(0, self.currentHealth - amount)
    self.healthChanged:Fire(self.currentHealth, self.maxHealth)
    if self.currentHealth <= 0 then
        self:die(damageType)
    end
end

function HealthModule:heal(amount)
    self.currentHealth = math.min(self.maxHealth, self.currentHealth + amount)
    self.healthChanged:Fire(self.currentHealth, self.maxHealth)
end

function HealthModule:die(damageType)
    print(self.player.Name .. " died from " .. (damageType or "unknown"))
end

return HealthModule

Inventory System

local Inventory = {}
Inventory.__index = Inventory

function Inventory.new(maxSlots)
    local self = setmetatable({}, Inventory)
    self.slots = {}
    self.maxSlots = maxSlots
    self.itemAdded = Instance.new("BindableEvent")
    for i = 1, maxSlots do
        self.slots[i] = nil
    end
    return self
end

function Inventory:addItem(itemId, quantity)
    for i, slot in ipairs(self.slots) do
        if slot and slot.id == itemId then
            slot.quantity = slot.quantity + quantity
            self.itemAdded:Fire(i, slot)
            return true
        end
    end
    for i, slot in ipairs(self.slots) do
        if not slot then
            self.slots[i] = {id = itemId, quantity = quantity}
            self.itemAdded:Fire(i, self.slots[i])
            return true
        end
    end
    return false
end

return Inventory

Player Data Manager

local PlayerData = {}
PlayerData.__index = PlayerData

function PlayerData.new(player, datastore)
    local self = setmetatable({}, PlayerData)
    self.player = player
    self.datastore = datastore
    self.data = {
        level = 1,
        experience = 0,
        coins = 0,
        inventory = {}
    }
    return self
end

function PlayerData:load()
    local success, data = pcall(function()
        return self.datastore:GetAsync("player_" .. self.player.UserId)
    end)
    if success and data then
        self.data = data
    end
end

function PlayerData:save()
    local success = pcall(function()
        self.datastore:SetAsync("player_" .. self.player.UserId, self.data)
    end)
    return success
end

return PlayerData

Experience & Leveling

local LevelSystem = {}
LevelSystem.__index = LevelSystem

function LevelSystem.new(player)
    local self = setmetatable({}, LevelSystem)
    self.player = player
    self.level = 1
    self.experience = 0
    self.levelChanged = Instance.new("BindableEvent")
    return self
end

function LevelSystem:addExperience(amount)
    self.experience = self.experience + amount
    local expNeeded = self.level * 100
    while self.experience >= expNeeded do
        self.experience = self.experience - expNeeded
        self:levelUp()
    end
end

function LevelSystem:levelUp()
    self.level = self.level + 1
    self.levelChanged:Fire(self.level)
    print(self.player.Name .. " is now level " .. self.level)
end

return LevelSystem
